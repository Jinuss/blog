# жҰӮи§Ҳ

vue3дёӯreactiveз”ЁдәҺе°Ҷжҷ®йҖҡеҜ№иұЎиҪ¬жҚўдёәе“Қеә”ејҸеҜ№иұЎпјҢе®ғзҡ„е®һзҺ°еҺҹзҗҶжҳҜйҖҡиҝҮProxyе’ҢReflectжқҘе®һзҺ°зҡ„гҖӮе…·дҪ“зҡ„е®һзҺ°ж–Үд»¶еҸӮи§Ғpackages\reactivity\src\reactive.tsгҖӮжң¬ж–Үдјҡд»Ӣз»Қreactiveзҡ„зӣёе…іapiеҰӮдёӢпјҡ

  • reactiveпјҡе°Ҷжҷ®йҖҡеҜ№иұЎиҪ¬жҚўдёәе“Қеә”ејҸеҜ№иұЎ
  • readonlyпјҡе°Ҷжҷ®йҖҡеҜ№иұЎиҪ¬жҚўдёәеҸӘиҜ»е“Қеә”ејҸеҜ№иұЎ
  • isReactiveпјҡеҲӨж–ӯдёҖдёӘеҜ№иұЎжҳҜеҗҰжҳҜе“Қеә”ејҸеҜ№иұЎ
  • isReadonlyпјҡеҲӨж–ӯдёҖдёӘеҜ№иұЎжҳҜеҗҰжҳҜеҸӘиҜ»е“Қеә”ејҸеҜ№иұЎ
  • isShallowпјҡеҲӨж–ӯдёҖдёӘеҜ№иұЎжҳҜеҗҰжҳҜжө…еұӮе“Қеә”ејҸеҜ№иұЎ
  • isProxyпјҡеҲӨж–ӯдёҖдёӘеҜ№иұЎжҳҜеҗҰжҳҜд»ЈзҗҶеҜ№иұЎ
  • shallowReactiveпјҡеҲӣе»әдёҖдёӘжө…еұӮе“Қеә”ејҸеҜ№иұЎ
  • shallowReadonlyпјҡеҲӣе»әдёҖдёӘжө…еұӮеҸӘиҜ»е“Қеә”ејҸеҜ№иұЎ
  • markRawпјҡж Үи®°дёҖдёӘеҜ№иұЎдёәеҺҹе§ӢеҜ№иұЎпјҢйҒҝе…Қиў«иҪ¬жҚўдёәе“Қеә”ејҸеҜ№иұЎ
  • toReadonlyпјҡе°ҶдёҖдёӘе“Қеә”ејҸеҜ№иұЎиҪ¬жҚўдёәеҸӘиҜ»е“Қеә”ејҸеҜ№иұЎ

function isReadonly(value) {

# createReactiveObject

reactiveзҡ„ж ёеҝғе®һзҺ°жҳҜcreateReactiveObjectеҮҪж•°пјҢе®ғзҡ„е®һзҺ°еҰӮдёӢпјҡ

function createReactiveObject(target, isReadonly2, baseHandlers, collectionHandlers, proxyMap) {
  // еҲӨж–ӯtargetжҳҜеҗҰжҳҜеҜ№иұЎпјҢиӢҘдёҚжҳҜпјҢеҲҷеј№еҮәиӯҰе‘ҠпјҢе№¶зӣҙжҺҘиҝ”еӣһ
  if (!isObject(target)) {
    {
      warn(
        `value cannot be made ${isReadonly2 ? "readonly" : "reactive"}: ${String(
          target
        )}`
      );
    }
    return target;
  }
  if (target["__v_raw"] && !(isReadonly2 && target["__v_isReactive"])) {
    return target;
  }
   const targetType = getTargetType(target);
  if (targetType === 0 /* INVALID */) {
    return target;
  }

  const existingProxy = proxyMap.get(target);
  if (existingProxy) {
    return existingProxy;
  }
  
  const proxy = new Proxy(
    target,
    targetType === 2 /* COLLECTION */ ? collectionHandlers : baseHandlers
  );
  proxyMap.set(target, proxy);
  return proxy;
}

function targetTypeMap(rawType) {
  switch (rawType) {
    case "Object":
    case "Array":
      return 1 /* COMMON */;
    case "Map":
    case "Set":
    case "WeakMap":
    case "WeakSet":
      return 2 /* COLLECTION */;
    default:
      return 0 /* INVALID */;
  }
}
function getTargetType(value) {
  return value["__v_skip"] || !Object.isExtensible(value) ? 0 /* INVALID */ : targetTypeMap(toRawType(value));
}

createReactiveObjectеҮҪж•°жҺҘеҸ—5дёӘеҸӮж•°пјҢдҫқж¬Ўдёәпјҡзӣ®ж ҮеҜ№иұЎtargetгҖҒжҳҜеҗҰеҸӘиҜ»isReadonly2гҖҒеҹәжң¬зұ»еһӢеӨ„зҗҶеҮҪж•°baseHandlersгҖҒйӣҶеҗҲеҜ№иұЎзҡ„д»ЈзҗҶеӨ„зҗҶеҮҪж•°collectionHandlersгҖҒе“Қеә”ејҸеҜ№иұЎзҡ„зј“еӯҳproxyMapгҖӮ

createReactiveObjectеңЁзЎ®дҝқtargetдёәеҜ№иұЎеҗҺпјҢдјҡжЈҖжөӢtargetжҳҜдёҖдёӘе“Қеә”ејҸеҜ№иұЎпјҢиӢҘжҳҜеҲҷзӣҙжҺҘиҝ”еӣһпјӣз„¶еҗҺдјҡд»Һзј“еӯҳproxyMapдёӯиҺ·еҸ–targetзҡ„д»ЈзҗҶеҜ№иұЎпјҢиӢҘеӯҳеңЁеҲҷзӣҙжҺҘиҝ”еӣһиҜҘд»ЈзҗҶеҜ№иұЎпјӣеҗҰеҲҷи°ғз”ЁgetTargetTypeеҮҪж•°еҲӨж–ӯзӣ®ж ҮеҜ№иұЎtargetзҡ„зұ»еһӢпјҢиӢҘзұ»еһӢдёәINVALIDпјҢеҲҷзӣҙжҺҘиҝ”еӣһtargetпјӣеҗҰеҲҷж №жҚ®зұ»еһӢеҲӣе»әд»ЈзҗҶеҜ№иұЎпјҢе№¶е°Ҷе…¶зј“еӯҳеҲ°proxyMapдёӯпјҢжңҖеҗҺиҝ”еӣһиҜҘд»ЈзҗҶеҜ№иұЎгҖӮ

getTargetTypeзҡ„е®һзҺ°д№ҹеңЁдёҠйқўпјҢиӢҘtargetеӯҳеңЁ__v_skipеұһжҖ§дё”дёәtrueпјҢжҲ–иҖ…targetдёҚжҳҜеҸҜжү©еұ•зҡ„еҜ№иұЎпјҢеҲҷиҝ”еӣһINVALIDпјӣеҗҰеҲҷж №жҚ®targetзҡ„зұ»еһӢи°ғз”ЁtargetTypeMapеҮҪж•°иҝ”еӣһзұ»еһӢгҖӮ

з”ұдёҠtargetTypeMapеҮҪж•°еҸҜзҹҘ,targetTypeMapеҮҪж•°ж №жҚ®targetзҡ„зұ»еһӢиҝ”еӣһдёҖдёӘж•°еӯ—пјҢеҲҶеҲ«иЎЁзӨәпјҡ

  • 0пјҡINVALIDпјҢиЎЁзӨәtargetдёҚжҳҜдёҖдёӘеҜ№иұЎ
  • 1пјҡCOMMONпјҢиЎЁзӨәtargetжҳҜдёҖдёӘжҷ®йҖҡеҜ№иұЎжҲ–ж•°з»„
  • 2пјҡCOLLECTIONпјҢиЎЁзӨәtargetжҳҜдёҖдёӘйӣҶеҗҲеҜ№иұЎ

з»јдёҠпјҢеҸҜд»ҘзҗҶи§ЈcreateReactiveObjectеҮҪж•°ж №жҚ®targetеҲӣе»әд»ЈзҗҶеҜ№иұЎзҡ„йҖ»иҫ‘еҰӮдёӢпјҡ

  • иӢҘtargetдёәжҷ®йҖҡеҜ№иұЎжҲ–ж•°з»„пјҢеҲҷеҲӣе»әжҷ®йҖҡеҜ№иұЎзҡ„д»ЈзҗҶеҜ№иұЎпјҢдҪҝз”ЁbaseHandlersеӨ„зҗҶеҮҪж•°пјӣ
  • иӢҘtargetдёәMap/Set/WeakMap/WeakSetпјҢеҲҷеҲӣе»әйӣҶеҗҲеҜ№иұЎзҡ„д»ЈзҗҶеҜ№иұЎпјҢдҪҝз”ЁcollectionHandlersеӨ„зҗҶеҮҪж•°пјӣ
  • е…¶дҪҷжғ…еҶөпјҢеҲҷзӣҙжҺҘиҝ”еӣһtargetгҖӮ

е…ідәҺеӨ„зҗҶеҷЁbaseHandlersе’ҢcollectionHandlersзҡ„е®һзҺ°пјҢдјҡеңЁеҗҺйқўзҡ„з« иҠӮдёӯд»Ӣз»ҚгҖӮ

# readonly/shallowReactive/shallowReadonly

function readonly(target) {
  return createReactiveObject(
    target,
    true, // иЎЁзӨәеҸӘиҜ»
    readonlyHandlers, // жҷ®йҖҡеҜ№иұЎзҡ„еҸӘиҜ»д»ЈзҗҶеӨ„зҗҶеҮҪж•°
    readonlyCollectionHandlers, // йӣҶеҗҲеҜ№иұЎзҡ„еҸӘиҜ»д»ЈзҗҶеӨ„зҗҶеҮҪж•°
    readonlyMap // еҸӘиҜ»е“Қеә”ејҸеҜ№иұЎзҡ„зј“еӯҳreadonlyMap

function shallowReactive(target) {
  return createReactiveObject(
    target,
    false, // дёҚжҳҜеҸӘиҜ»
    shallowReactiveHandlers, // жҷ®йҖҡеҜ№иұЎзҡ„жө…еұӮе“Қеә”ејҸд»ЈзҗҶеӨ„зҗҶеҮҪж•°

    shallowReactiveMap // жө…еұӮе“Қеә”ејҸеҜ№иұЎзҡ„зј“еӯҳshallowReactiveMap
  );
}

function shallowReadonly(target) {

  );
}

# markRaw

markRawз”ЁдәҺж Үи®°еҜ№иұЎдёәеҺҹе§ӢеҜ№иұЎпјҢиҝҷз§ҚеҜ№иұЎдёҚдјҡиў«иҪ¬жҚўдёәе“Қеә”ејҸеҜ№иұЎпјҢд№ҹдёҚдјҡиў«д»ЈзҗҶпјҢе…¶е®һзҺ°еҰӮдёӢпјҡ

function markRaw(value) {
  if (!hasOwn(value, "__v_skip") && Object.isExtensible(value)) {
    def(value, "__v_skip", true);
  }
  return value;
}
