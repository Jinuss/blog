# CanvasVectorLayerRenderer жёІжҹ“еҷЁ

VectorLayerзұ»дёӯcreateRendererж–№жі•е®һйҷ…жҳҜйҮҚеҶҷLayerзұ»пјҲеҚіBaseVectorLayerзҡ„зҲ¶зұ»пјүдёӯе®ҡд№үзҡ„еҗҢеҗҚж–№жі•пјҢиҝ”еӣһдёҖдёӘCanvasVectorLayerRendererзұ»зҡ„е®һдҫӢеҜ№иұЎпјҢиҜҘзұ»з»§жүҝдәҶCanvasLayerRendererзұ»гҖӮ

CanvasVectorLayerRendererзұ»зҡ„жһ„йҖ еҮҪж•°еҗ„еҸҳйҮҸеҗ«д№үеҰӮдёӢйқўд»Јз ҒзүҮж®өжүҖзӨә

# CanvasVectorLayerRendererж–№жі•и§ЈйҮҠ

  • renderWorldsж–№жі•

renderWorldsж–№жі•з”ЁдәҺеңЁеӨҡдёӘдё–з•ҢдёҠжёІжҹ“зҹўйҮҸж•°жҚ®,еҚіең°еӣҫеңЁж°ҙе№іиҪҙдёҠеҸҜд»ҘеҫӘзҺҜе»¶з”ігҖӮе®ғжҺҘеҸ—дёүдёӘеҸӮж•°executorGroup(иҙҹиҙЈжёІжҹ“жү§иЎҢзҡ„еҜ№иұЎпјҢиҝӣиЎҢе®һйҷ…жёІжҹ“зҡ„ж“ҚдҪң),frameState(её§зҠ¶жҖҒдҝЎжҒҜ),declutterable(иЎЁзӨәжҳҜеҗҰеҸӮдёҺеҺ»йҮҚпјҢtrueйңҖиҰҒеҺ»йҮҚпјҢfalseдёҚиҝӣиЎҢеҺ»йҮҚеӨ„зҗҶ)

renderWorldsж–№жі•еҶ…йғЁе°ұжҳҜе…ҲиҺ·еҸ–её§зҠ¶жҖҒзҡ„еҗ„з§ҚдҝЎжҒҜдёӯеҝғгҖҒеҲҶиҫЁзҺҮгҖҒеғҸзҙ жҜ”зӯүзӯүпјҢз„¶еҗҺи°ғз”Ёthis.getLayerиҺ·еҸ–еӣҫеұӮжәҗд»ҘеҸҠжҠ•еҪұеқҗж Үзі»жҳҜеҗҰж”ҜжҢҒеңЁж°ҙе№іж–№еҗ‘е»¶з”іпјҢд»ҺиҖҢзЎ®е®ҡжёІжҹ“зҡ„иө·е§ӢдҪҚзҪ®пјҢжңҖеҗҺжү§иЎҢдёҖдёӘеҫӘзҺҜпјҢж №жҚ®еҪ“еүҚи§Ҷеӣҫзҡ„дё–з•ҢиҢғеӣҙпјҢйҖҗдёӘжёІжҹ“жҜҸдёӘдё–з•Ң.

renderWorldsж–№жі•дёӯзҡ„ж ёеҝғйғЁеҲҶеңЁеҫӘзҺҜдҪ“пјҢиҜҘеҫӘзҺҜдҪ“е…ҲжҳҜи°ғз”ЁLayerRendererзұ»пјҲеҚіCanvasLayerRendererзұ»зҡ„зҲ¶зұ»пјүдёӯзҡ„getRenderTransformж–№жі•иҺ·еҸ–жёІжҹ“иҪ¬жҚўзҹ©йҳөпјҢе®ғдјҡж №жҚ®еҪ“еүҚи§Ҷеӣҫзҡ„дёӯеҝғгҖҒеҲҶиҫЁзҺҮгҖҒж—ӢиҪ¬и§’еәҰзӯүдҝЎжҒҜпјҢи®Ўз®—еҮәеҪ“еүҚдё–з•Ңзҡ„еҸҳжҚўзҹ©йҳөгҖӮз„¶еҗҺж №жҚ®frameState.declutter)зЎ®е®ҡеҸҳжҚўзҹ©йҳөпјҢеҶҚйҖҡиҝҮexecutorGroup.executeи§ҰеҸ‘е®һйҷ…зҡ„жёІжҹ“ж“ҚдҪңпјҢиҝҷжҳҜдёҖдёӘеҫҲеӨҚжқӮзҡ„иҝҮзЁӢпјҢеңЁд»ҘеҗҺдјҡеҲҶжһҗеҲ°гҖӮ

renderWorlds(executorGroup, frameState, declutterable) {
    const extent = frameState.extent;
    const viewState = frameState.viewState;
    const center = viewState.center;
    const resolution = viewState.resolution;
    const projection = viewState.projection;
    const rotation = viewState.rotation;
    const projectionExtent = projection.getExtent();
    const vectorSource = this.getLayer().getSource();
    const declutter = this.getLayer().getDeclutter();
    const pixelRatio = frameState.pixelRatio;
    const viewHints = frameState.viewHints;
    const snapToPixel = !(
      viewHints[ViewHint.ANIMATING] || viewHints[ViewHint.INTERACTING]
    );
    const context = this.context;
    const width = Math.round((getWidth(extent) / resolution) * pixelRatio);//жёІжҹ“еҢәеҹҹе®ҪеәҰ
    const height = Math.round((getHeight(extent) / resolution) * pixelRatio);//жёІжҹ“еҢәеҹҹй«ҳеәҰ

    const multiWorld = vectorSource.getWrapX() && projection.canWrapX();//жҳҜеҗҰж°ҙе№іж–№еҗ‘е»¶з”і
    const worldWidth = multiWorld ? getWidth(projectionExtent) : null; //дё–з•Ңзҡ„е®ҪеәҰ
    const endWorld = multiWorld
      ? Math.ceil((extent[2] - projectionExtent[2]) / worldWidth) + 1
      : 1; //и®Ўз®—йңҖиҰҒжёІжҹ“дё–з•Ңзҡ„ж•°йҮҸ
    let world = multiWorld
      ? Math.floor((extent[0] - projectionExtent[0]) / worldWidth)
      : 0;//еҪ“еүҚи§ҶеӣҫжүҖеңЁзҡ„дё–з•Ңзҡ„иө·е§ӢдҪҚзҪ®пјҲеҚіејҖе§ӢжёІжҹ“зҡ„дё–з•Ңзј–еҸ·пјү
    do {
      let transform = this.getRenderTransform(
        center,
        resolution,
        0,
        pixelRatio,
        width,
        height,
        world * worldWidth
      );
      if (frameState.declutter) {
        transform = transform.slice(0);
      }
      executorGroup.execute(
        context,
        [context.canvas.width, context.canvas.height],
        transform,
        rotation,
        snapToPixel,
        declutterable === undefined
          ? ALL
          : declutterable
          ? DECLUTTER
          : NON_DECLUTTER,
        declutterable ? declutter && frameState.declutter[declutter] : undefined
      );
    } while (++world < endWorld);
  }

setDrawContext_() {
    if (this.opacity_ !== 1) {
      this.targetContext_ = this.context;
      this.context = createCanvasContext2D(
        this.context.canvas.width,
        this.context.canvas.height,
        canvasPool,
      );
    }
  }

  • е’ҢresetDrawContext_ж–№жі•

resetDrawContext_ж–№жі•е°ұжҳҜйҮҚзҪ®canvasзҡ„дёҠдёӢж–ҮзҺҜеўғжҲ–зҠ¶жҖҒпјҢreleaseCanvasе°ұжҳҜйҮҠж”ҫз”»еёғиө„жәҗпјҢжё…з©әcanvasдёҠзҡ„ж•°жҚ®пјҢд»Ҙдҫҝд»ҘеҗҺеӨҚз”ЁжҲ–еӣһж”¶гҖӮ

  resetDrawContext_() {
    if (this.opacity_ !== 1) {
      const alpha = this.targetContext_.globalAlpha;
      this.targetContext_.globalAlpha = this.opacity_;
      this.targetContext_.drawImage(this.context.canvas, 0, 0);
      this.targetContext_.globalAlpha = alpha;
      releaseCanvas(this.context);
      canvasPool.push(this.context.canvas);
      this.context = this.targetContext_;
      this.targetContext_ = null;
    }
  }
