жң¬ж–Үдё»иҰҒд»Ӣз»ҚMultiLineStringзұ»зҡ„жәҗз Ғе®һзҺ°е’ҢеҺҹзҗҶгҖӮ

# MultiLineStringзұ»зҡ„жәҗз Ғе®һзҺ°

class MultiLineString extends SimpleGeometry {
  constructor(coordinates, layout, ends) {
    super();
    this.ends_ = [];
    this.maxDelta_ = -1;
    this.maxDeltaRevision_ = -1;

    if (Array.isArray(coordinates[0])) {
      this.setCoordinates(coordinates, layout);
    } else if (layout !== undefined && ends) {
      this.setFlatCoordinates(layout, coordinates);
      this.ends_ = ends;
    } else {
      const lineStrings = coordinates;
      const flatCoordinates = [];
      const ends = [];
      for (let i = 0, ii = lineStrings.length; i < ii; ++i) {
        const lineString = lineStrings[i];
        extend(flatCoordinates, lineString.getFlatCoordinates());
        ends.push(flatCoordinates.length);
      }
      const layout =
        lineStrings.length === 0
          ? this.getLayout()
          : lineStrings[0].getLayout();
      this.setFlatCoordinates(layout, flatCoordinates);
      this.ends_ = ends;
    }
  }
  appendLineString(lineString) {
    extend(this.flatCoordinates, lineString.getFlatCoordinates().slice());
    this.ends_.push(this.flatCoordinates.length);
    this.changed();
  }
  clone() {
    const multiLineString = new MultiLineString(
      this.flatCoordinates.slice(),
      this.layout,
      this.ends_.slice()
    );
    multiLineString.applyProperties(this);
    return multiLineString;
  }
  closestPointXY(x, y, closestPoint, minSquaredDistance) {
    if (minSquaredDistance < closestSquaredDistanceXY(this.getExtent(), x, y)) {
      return minSquaredDistance;
    }
    if (this.maxDeltaRevision_ != this.getRevision()) {
      this.maxDelta_ = Math.sqrt(
        arrayMaxSquaredDelta(
          this.flatCoordinates,
          0,
          this.ends_,
          this.stride,
          0
        )
      );
      this.maxDeltaRevision_ = this.getRevision();
    }
    return assignClosestArrayPoint(
      this.flatCoordinates,
      0,
      this.ends_,
      this.stride,
      this.maxDelta_,
      false,
      x,
      y,
      closestPoint,
      minSquaredDistance
    );
  }
  getCoordinateAtM(m, extrapolate, interpolate) {
    if (
      (this.layout != "XYM" && this.layout != "XYZM") ||
      this.flatCoordinates.length === 0
    ) {
      return null;
    }
    extrapolate = extrapolate !== undefined ? extrapolate : false;
    interpolate = interpolate !== undefined ? interpolate : false;
    return lineStringsCoordinateAtM(
      this.flatCoordinates,
      0,
      this.ends_,
      this.stride,
      m,
      extrapolate,
      interpolate
    );
  }
  getCoordinates() {
    return inflateCoordinatesArray(
      this.flatCoordinates,
      0,
      this.ends_,
      this.stride
    );
  }
  getEnds() {
    return this.ends_;
  }
  getLineString(index) {
    if (index < 0 || this.ends_.length <= index) {
      return null;
    }
    return new LineString(
      this.flatCoordinates.slice(
        index === 0 ? 0 : this.ends_[index - 1],
        this.ends_[index]
      ),
      this.layout
    );
  }
  getFlatMidpoints() {
    const midpoints = [];
    const flatCoordinates = this.flatCoordinates;
    let offset = 0;
    const ends = this.ends_;
    const stride = this.stride;
    for (let i = 0, ii = ends.length; i < ii; ++i) {
      const end = ends[i];
      const midpoint = interpolatePoint(
        flatCoordinates,
        offset,
        end,
        stride,
        0.5
      );
      extend(midpoints, midpoint);
      offset = end;
    }
    return midpoints;
  }
  getSimplifiedGeometryInternal(squaredTolerance) {
    const simplifiedFlatCoordinates = [];
    const simplifiedEnds = [];
    simplifiedFlatCoordinates.length = douglasPeuckerArray(
      this.flatCoordinates,
      0,
      this.ends_,
      this.stride,
      squaredTolerance,
      simplifiedFlatCoordinates,
      0,
      simplifiedEnds
    );
    return new MultiLineString(simplifiedFlatCoordinates, "XY", simplifiedEnds);
  }
  getType() {
    return "MultiLineString";
  }
  intersectsExtent(extent) {
    return intersectsLineStringArray(
      this.flatCoordinates,
      0,
      this.ends_,
      this.stride,
      extent
    );
  }
  setCoordinates(coordinates, layout) {
    this.setLayout(layout, coordinates, 2);
    if (!this.flatCoordinates) {
      this.flatCoordinates = [];
    }
    const ends = deflateCoordinatesArray(
      this.flatCoordinates,
      0,
      coordinates,
      this.stride,
      this.ends_
    );
    this.flatCoordinates.length = ends.length === 0 ? 0 : ends[ends.length - 1];
    this.changed();
  }
}

# MultiLineStringзұ»зҡ„жһ„йҖ еҮҪж•°

MultiLineStringзұ»зҡ„жһ„йҖ еҮҪж•°жҺҘеҸ—дёүдёӘеҸӮж•°:еқҗж Үж•°з»„coordinatesгҖҒеқҗж ҮеёғеұҖж–№ејҸlayoutе’ҢжҜҸдёӘзәҝж®өзҡ„з»“жқҹзӮ№endsпјӣжһ„йҖ еҮҪж•°еҶ…йғЁиҝҳе®ҡд№үдәҶдёӨдёӘеҸҳйҮҸthis.maxDelta_е’Ңthis.maxDeltaRevision_з”ЁдәҺи®°еҪ•еҮ дҪ•еҜ№иұЎзҡ„дҝ®и®ўзүҲжң¬еҸ·пјҢеңЁеҮ дҪ•еҜ№иұЎиҝӣиЎҢйҮҚж–°и®Ўз®—е’ҢжёІжҹ“ж—¶з”ЁдәҺеҲӨж–ӯжҳҜеҗҰдҪҝз”Ёзј“еӯҳпјҢеҸҜд»ҘжҸҗй«ҳжҖ§иғҪпјҢе®ғд»¬еҲқе§ӢеҖјйғҪжҳҜдёә-1;жһ„йҖ еҮҪж•°дјҡеҲӨж–ӯеҸӮж•°coordinatesжҳҜеҗҰдёәдёҖдёӘдәҢз»ҙж•°з»„пјҢиӢҘжҳҜпјҢеҲҷи°ғз”Ёthis.setCoordinatesж–№жі•и®ҫзҪ®еқҗж ҮпјӣеҗҰеҲҷпјҢеҲӨж–ӯеҸӮж•°layoutе’ҢendsпјҢиӢҘе®ғд»¬йғҪдј еҖјдәҶпјҢеҲҷи°ғз”Ёthis.setFlatCoordinatesж–№жі•пјӣеҗҰеҲҷи®ӨдёәcoordinatesиЎЁзӨәзҡ„жҳҜеӨҡдёӘзәҝж®өзҡ„ж•°з»„пјӣз„¶еҗҺеҫӘзҺҜйҒҚеҺҶе®ғпјҢе°ҶжҜҸдёӘзәҝж®өзҡ„еқҗж Үж•°жҚ®ж·»еҠ еҲ°еҸҳйҮҸflatCoordinatesдёӯпјҢе№¶дё”е°ҶжҜҸдёӘзәҝж®өзҡ„еқҗж Үй•ҝеәҰж·»еҠ еҲ°endsдёӯпјҢжңҖеҗҺи°ғз”Ёthis.setFlatCoordinatesж–№жі•пјҢи®ҫзҪ®еқҗж ҮгҖӮ

# MultiLineStringзұ»зҡ„дё»иҰҒж–№жі•

MultiLineStringзұ»зҡ„дё»иҰҒж–№жі•еҰӮдёӢпјҡ

  • appendLineStringж–№жі•пјҡеҗ‘еҪ“еүҚеҮ дҪ•еҜ№иұЎдёӯж·»еҠ дёҖжқЎзәҝж®өпјҢеҸӮж•°lineStringжҳҜLineStringзұ»зҡ„е®һдҫӢпјҢеҶ…йғЁе°ұжҳҜдјҡи°ғз”ЁlineString.getFlatCoordinatesиҺ·еҸ–еҸӮж•°зәҝж®өзҡ„еқҗж ҮпјҢз„¶еҗҺе°Ҷе®ғж·»еҠ еҲ°this.flatCoordinatesдёӯпјҢжӣҙж–°this.ends_зҡ„еҖјпјҢжңҖеҗҺи°ғз”Ёthis.changedж–№жі•

  • cloneж–№жі•пјҡеӨҚеҲ¶еҮ дҪ•еҜ№иұЎпјҢе®һдҫӢеҢ–MultiLineStringзұ»пјҢи°ғз”ЁapplyPropertiesж–№жі•еә”з”ЁеұһжҖ§пјҢжңҖеҗҺиҝ”еӣһе®һдҫӢеҜ№иұЎгҖӮ

  • closestPointXYж–№жі•: жҺҘеҸ—еӣӣдёӘеҸӮж•°пјҢз»ҷе®ҡзӮ№зҡ„xгҖҒyгҖҒжңҖиҝ‘зӮ№closestPointе’ҢжңҖе°Ҹи·қзҰ»е№іж–№minSquaredDistance;closestPointXYж–№жі•еҶ…йғЁдјҡе…Ҳи°ғз”ЁclosestSquaredDistanceXYи®Ўз®—з»ҷе®ҡзӮ№дёҺеҢ…еӣҙзӣ’зҡ„жңҖе°Ҹи·қзҰ»зҡ„е№іж–№пјҢе°Ҷд№ӢдёҺminSquaredDistanceзҡ„еӨ§е°ҸиҝӣиЎҢеҜ№жҜ”пјҢиӢҘminSquaredDistanceиҫғе°ҸпјҢеҲҷиҝ”еӣһе®ғпјӣеҗҰеҲҷпјҢеҲӨж–ӯеҮ дҪ•еҜ№иұЎжҳҜеҗҰеҸ‘з”ҹдәҶж”№еҸҳпјҢиӢҘжңӘж”№еҸҳпјҢеҲҷдҪҝз”Ёд№ӢеүҚзҡ„this.maxDelta_и°ғз”ЁassignClosestArrayPointж–№жі•и®Ўз®—жңҖе°Ҹи·қзҰ»е№іж–№е№¶иҝ”еӣһпјҢе№¶дё”жӣҙж–°жңҖиҝ‘зӮ№еқҗж ҮclosestPoint;иӢҘж”№еҸҳдәҶпјҢеҲҷи°ғз”ЁarrayMaxSquaredDeltaж–№жі•и®Ўз®—еҮ дҪ•еҜ№иұЎзӣёйӮ»дёӨзӮ№еқҗж Үзҡ„жңҖеӨ§и·қзҰ»пјҢе°Ҷе…¶иөӢеҖјз»ҷthis.maxDelta_пјҢжӣҙж–°this.maxDeltaRevision_зҡ„еҖјпјҢжңҖеҗҺи°ғз”ЁassignClosestArrayPointж–№жі•гҖӮ

  • getCoordinateAtMж–№жі•пјҡдҪҝз”ЁзәҝжҖ§жҸ’еҖјиҝ”еӣһеҮ дҪ•еҜ№иұЎеңЁеұһжҖ§дёәMеӨ„зҡ„еқҗж ҮпјҢеҰӮжһңдёҚеӯҳеңЁпјҢеҲҷиҝ”еӣһnull e

  • getCoordinatesж–№жі•пјҡиҝ”еӣһеқҗж Үж•°жҚ®пјҢеҶ…йғЁе°ұжҳҜи°ғз”ЁinflateCoordinatesArrayж–№жі•е°ҶжүҒе№іеҢ–дёҖз»ҙж•°з»„this.flatCoordinatesиҪ¬дёәеӨҡз»ҙж•°з»„е№¶иҝ”еӣһгҖӮ

  • getEndsж–№жі•пјҡиҝ”еӣһthis.endsеҖјпјҢеҚіиҝ”еӣһжүҖжңүзәҝж®өзҡ„з»“жқҹзӮ№

  • getLineStringж–№жі•пјҡжҺҘеҸ—дёҖдёӘеҸӮж•°indexпјҢеҲӨж–ӯеҸӮж•°indexжҳҜеҗҰеҗҲжі•пјӣз„¶еҗҺжҲӘеҸ–this.flatCoordinatesе®һдҫӢеҢ–LineStringзұ»е№¶иҝ”еӣһе®һдҫӢеҜ№иұЎгҖӮ

  • getLineStringsж–№жі•пјҡиҝ”еӣһеҮ дҪ•еҜ№иұЎзҡ„жүҖжңүзәҝж®өпјҢйҖҡиҝҮйҒҚеҺҶthis.endsжҲӘеҸ–this.flatCoordinatesпјҢз„¶еҗҺе®һдҫӢеҢ–LineStringзұ»пјҢжңҖеҗҺиҝ”еӣһеҢ…еҗ«жүҖжңүзәҝж®өзҡ„lineStringsж•°з»„гҖӮ

  • getFlatMidpointsж–№жі•пјҡиҝ”еӣһжүҖжңүзәҝж®өзҡ„дёӯзӮ№пјҢж ёеҝғж–№жі•жҳҜinterpolatePointиҺ·еҸ–жҜҸжқЎзәҝж®өзҡ„дёӯзӮ№пјҢз„¶еҗҺе°Ҷе…¶ж”ҫе…Ҙж•°з»„midpointsдёӯпјҢжңҖеҗҺиҝ”еӣһгҖӮ

  • getSimplifiedGeometryInternalж–№жі•пјҡиҜҘж–№жі•з”ЁдәҺиҺ·еҸ–еҮ дҪ•еҜ№иұЎзҡ„з®ҖеҢ–зүҲпјҢеҶ…йғЁд№ҹжҳҜи°ғз”ЁdouglasPeuckerз®—жі•иҝӣиЎҢз®ҖеҢ–еҮ дҪ•еҜ№иұЎпјҢиҺ·еҸ–з®ҖеҢ–еҜ№иұЎзҡ„еқҗж Үж•°жҚ®пјҢз„¶еҗҺдј е…ҘиҺ·еҸ–зҡ„еқҗж Үж•°жҚ® е®һдҫӢеҢ–MultiLineStringзұ»пјҢжңҖеҗҺиҝ”еӣһе®һдҫӢгҖӮ

  • getTypeж–№жі•пјҡиҝ”еӣһеҮ дҪ•еҜ№иұЎзҡ„зұ»еһӢпјҢMultiLineString

  • intersectsExtentж–№жі•пјҡеҲӨж–ӯзҹ©еҪўextentжҳҜеҗҰдёҺеҮ дҪ•еҜ№иұЎзӣёдәӨпјҢеҶ…йғЁи°ғз”Ёзҡ„жҳҜintersectsLineStringArrayж–№жі•е№¶иҝ”еӣһе…¶з»“жһңпјӣ

  • setCoordinatesж–№жі•пјҡи®ҫзҪ®еқҗж Үthis.flatCoordinatesпјҢеқҗж ҮеёғеұҖж–№ејҸthis.layoutд»ҘеҸҠжӯҘе№…this.stride
